#include <iostream>
using namespace std;
int main() {
    int meuVetor[] = {21, 7, -21, -9, 76, -17, 41};
    
    int s1 = meuVetor[5];
    int s2 = meuVetor[3];

    while (s1 < s2) {
        s1 += 3;
        s2 -= 2;
    }

    meuVetor[1] = s1 + s2;
    meuVetor[4] = s1;
    meuVetor[5] = s2;

    for (int i = 0; i < 7; i++)
        cout << "meuVetor["<< i << "]: " << meuVetor[i] << endl;

    return 0;
}
